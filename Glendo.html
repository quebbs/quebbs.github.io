X=linspace(4632,4642, 250)
Den=norm.pdf(X,4637,1)
I=array([0. if X[i]<=4637 else (X[i]-4637.) for i in range(len(X))])

plt.figure(1,figsize=(10,3))
plt.subplot(1,2,1)
plt.plot(X,Den)
plt.xlabel('Water Level (ft)')
plt.ylabel('Density')
plt.twinx()
plt.plot(X,I,c='g',ls='--')
plt.ylabel('Impact ($)')

E=[]
for Di,Ii in zip(Den,I):
    E.append(Di*Ii*(4642-4632.)/len(X))
plt.subplot(1,2,2)
plt.plot(X,E,c='r')
plt.ylabel('Product')
plt.xlabel('Water Level (ft)')
plt.text(4633.2,0.0042,'E[Impact]='+str(round(sum(E),3)))
print sum(E)
plt.tight_layout()
plt.savefig('Glendo_Density.png')
